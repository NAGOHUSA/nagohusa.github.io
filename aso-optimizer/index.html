async function handleOptimize(request, env) { let body; try { body = await request.json(); } catch { return errorResponse('Request body must be valid JSON'); } const { metadata, sessionId, paymentVerified } = body ?? {}; if (!metadata || typeof metadata !== 'object') { return errorResponse('metadata object is required'); } const markets = getMarketRecommendations(metadata.primaryGenre); // Free tier: return only market recommendations (no AI call) if (!sessionId && !paymentVerified) { return jsonResponse({ markets, premium: false }); } // Premium tier: verify payment let isPaid = false; if (sessionId) { try { const stripeRes = await fetch(`https://api.stripe.com/v1/checkout/sessions/${encodeURIComponent(sessionId)}`, { headers: { 'Authorization': `Bearer ${env.STRIPE_SECRET_KEY}` }, }); if (stripeRes.ok) { const session = await stripeRes.json(); isPaid = session.payment_status === 'paid'; } } catch (err) { console.error('Stripe verification error:', err); isPaid = false; } } else if (paymentVerified === true) { isPaid = true; } if (!isPaid) { return errorResponse('Payment not confirmed. Complete checkout before accessing premium results.', 402); } // Generate reports for ALL markets in parallel const allReports = []; for (const market of markets) { const locale = market.locale; const marketName = market.market; // Call DeepSeek for each market let aiResponse; try { const deepseekRes = await fetch('https://api.deepseek.com/v1/chat/completions', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${env.DEEPSEEK_API_KEY}`, }, body: JSON.stringify({ model: 'deepseek-chat', messages: [ { role: 'system', content: 'You are an expert App Store Optimization specialist. Always return valid JSON only — no markdown, no extra text.', }, { role: 'user', content: buildASOPrompt(metadata, locale, marketName), }, ], temperature: 0.6, max_tokens: 4000, }), }); if (!deepseekRes.ok) { console.error(`DeepSeek API error for ${marketName}:`, deepseekRes.status); continue; } aiResponse = await deepseekRes.json(); const aiContent = aiResponse.choices?.[0]?.message?.content ?? ''; const optimized = parseAIResponse(aiContent); allReports.push({ market: marketName, locale: locale, flag: market.flag, ...optimized }); } catch (err) { console.error(`DeepSeek fetch error for ${marketName}:`, err); } } return jsonResponse({ markets, reports: allReports, premium: true, }); }
